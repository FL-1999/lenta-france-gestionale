# ============================================================
# 1) Base image con Python
# ============================================================
FROM python:3.11-slim

# Non mostrare messaggi non necessari
ENV PYTHONUNBUFFERED=1

# Cartella di lavoro nel container
WORKDIR /app

# ============================================================
# 2) Copia i file nellâ€™immagine
# ============================================================
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY app.py .

# ============================================================
# 3) Espone la porta (FastAPI lavora su 8000)
# ============================================================
EXPOSE 8000

# ============================================================
# 4) Comando di avvio (versione professionale)
# ============================================================
CMD ["uvicorn", "app:app", "--host", "0.0.0.0", "--port", "8000"]
